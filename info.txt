              document.addEventListener("DOMContentLoaded", function () {
        const chart = document.querySelector(
          ".AccuracyChart-module--chart--5e9ea",
        );
        const modal = document.querySelector(
          ".AccuracyChart-module--card--1ff00",
        );
        const fingerprintBall = document.querySelector(
          ".AccuracyChart-module--fingerPoint--57835",
        );
        const competitorBall = document.querySelector(
          ".AccuracyChart-module--compPoint--b433d",
        );
        const fingerprintValue = document.querySelector(
          ".AccuracyChart-module--finValue--6c97d",
        );
        const competitorValue = document.querySelector(
          ".AccuracyChart-module--comValue--b5ee0",
        );
        const daysLabel = document.querySelector(
          ".AccuracyChart-module--days--4391d",
        );
        const activeLine = document.querySelector(
          ".AccuracyChart-module--lineActiveX--80445",
        );
        const hoverArea = document.querySelector(
          ".AccuracyChart-module--hoverArea--9b490",
        );

        const totalWidth = chart.clientWidth;
        const totalHeight = chart.clientHeight;
        const centerOffset = totalHeight * 0.2; // Adjust to keep balls near the curve center
        const xAxisStart =
          activeLine.getBoundingClientRect().left -
          chart.getBoundingClientRect().left;
        const dates = Array.from({ length: 120 }, (_, i) => i + 1);

        const fingerprintCurve = (x) =>
          totalHeight * (0.25 - 0.04 * Math.sin(x * Math.PI)) - centerOffset;
        const competitorCurve = (x) =>
          totalHeight * (0.6 - 0.32 * Math.cos(x * Math.PI)) - centerOffset;

        fingerprintBall.style.left = `${xAxisStart}px`;
        fingerprintBall.style.top = `${fingerprintCurve(0)}px`;
        competitorBall.style.left = `${xAxisStart}px`;
        competitorBall.style.top = `${competitorCurve(0)}px`;
        console.log("Fingerprint Ball:", fingerprintBall);
        console.log("Competitor Ball:", competitorBall);

        chart.addEventListener("mouseenter", function () {
          modal.classList.add("AccuracyChart-module--active--d51ee");
          fingerprintBall.style.display = "block";
          competitorBall.style.display = "block";
        });

        chart.addEventListener("mouseleave", function () {
          modal.classList.remove("AccuracyChart-module--active--d51ee");
          fingerprintBall.style.display = "none";
          competitorBall.style.display = "none";
        });

        chart.addEventListener("mousemove", function (event) {
           console.log("Fingerprint Ball:", fingerprintBall);
        console.log("Competitor Ball:", competitorBall);
          const rect = chart.getBoundingClientRect();
          let x = event.clientX - rect.left;
          if (x < xAxisStart) return;

          const percentage =
            ((x - xAxisStart) / (totalWidth - xAxisStart)) * 100;
          const index = Math.floor(
            (dates.length * (x - xAxisStart)) / (totalWidth - xAxisStart),
          );

          if (index >= 0 && index < dates.length) {
            const days = dates[index];
            const normalizedX = index / 120;

            fingerprintValue.textContent =
              (-1.4 * normalizedX).toFixed(1) + "%";
            competitorValue.textContent =
              (-75.8 * normalizedX).toFixed(1) + "%";
            daysLabel.textContent = days + " Days";

            activeLine.style.left = `${x}px`;
            fingerprintBall.style.left = `${x}px`;
            competitorBall.style.left = `${x}px`;
            fingerprintBall.style.top = `${fingerprintCurve(normalizedX)}px`;
            competitorBall.style.top = `${competitorCurve(normalizedX)}px`;

            hoverArea.classList.add("AccuracyChart-module--active--d51ee");
            modal.style.left = `calc(${x}px - 80px)`; // Shift modal left so it doesn't cover the balls
            modal.style.transform = "translateX(-50%)";
          }
        });
      });
